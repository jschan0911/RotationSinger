import datetime as dt rotationlist = ['박소윤, 이다영', '박세윤, 진승휘'] fixedlist = '정승찬' targetday = dt.date(2022,6,12) today = dt.datetime.now().date() day = (today - targetday).days if (day % 7 != 0): day += 7 rotation = int((day/7)%len(rotationlist)) singers = rotationlist[rotation] + ', ' + fixedlist print('이번주 대예배 싱어는 ' + singers + '입니다.')
